#!/bin/bash
echo "Enter a number: " 
read number
reverse=0 
original=$number
while [ $number -ne 0 ] 
do
      remainder=$(( $number % 10 )) 
      reverse=$(( $reverse * 10 + $remainder )) 
      number=$(( $number / 10 ))
done
if [ $original -eq $reverse ] 
then
      echo "$original is a palindrome." 
else
      echo "$original is not a palindrome."
fi

#!/bin/bash
# Prompt user for input string
read -p "Enter a string: " input_string

# Convert to lowercase to count both uppercase and lowercase vowels
lower_string=$(echo "$input_string" | tr 'A-Z' 'a-z')

# Count vowels using grep and wc
vowel_count=$(echo "$lower_string" | grep -o '[aeiou]' | wc -l)
echo "Number of vowels in the given string: $vowel_count”

#!/bin/bash
while true
do
echo "======================================="
 echo " Zombie and Orphan Process Demo"
 echo "======================================="
 echo "1. Demonstrate Zombie Process"
 echo "2. Demonstrate Orphan Process"
 echo "3. Exit"
 echo "======================================="
 read -p "Enter your choice: " choice
 case $choice in
 1)
 echo ""
 echo "Creating Zombie Process..."
 echo "------------------------------"
 # Fork a background child process that exits immediately
 (
 echo "Child (Zombie) PID: $$"
 exit 0
 ) &
 # Get the PID of the background process
 zombie_pid=$!
 echo "Parent Process PID: $$"
 echo "Waiting before reaping child..."
 # Sleep to give time for child to exit and become zombie
 sleep 5
echo "Checking process list (look for Z status)..."
 ps -l | grep $zombie_pid
echo "Now reaping the zombie..."
 wait $zombie_pid
 echo "Zombie reaped."
 echo ""
 ;;
 2)
 echo ""
 echo "Creating Orphan Process..."
 echo "------------------------------"
 (
 echo "Child (Orphan) PID: $$"
 sleep 10
 echo "Child (Orphan) process's new PPID: $(ps -o ppid= -p $$)"
 ) &
 echo "Parent process exiting immediately..."
 exit
 ;;
 3)
 echo "Exiting..."
 break
 ;;
 *)
 echo "Invalid choice. Please try again."
 ;;
 esac
echo ""
 read -p "Press Enter to continue..."
 clear
done
